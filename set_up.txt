cd F:\GSter

# 2) 创建环境（Python 3.10）

conda run -n gstar310 python -m pip install --upgrade pip

conda run -n gstar310 python -c "import numpy,sklearn;print('[OK]',numpy.__version__,sklearn.__version__)"
